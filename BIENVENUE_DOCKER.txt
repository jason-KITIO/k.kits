🚀 DÉMARRAGE RAPIDE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Windows :
  1. copy .env.docker .env
  2. .\docker.ps1 up
  3. start http://localhost:3000

Linux/Mac :
  1. cp .env.docker .env
  2. make up  (ou ./docker.sh up)
  3. open http://localhost:3000

🔧 COMMANDES UTILES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Windows PowerShell :
  .\docker.ps1 help      - Voir toutes les commandes
  .\docker.ps1 up        - Démarrer
  .\docker.ps1 down      - Arrêter
  .\docker.ps1 logs      - Voir les logs
  .\docker.ps1 status    - État des services

Linux/Mac :
  make help              - Voir toutes les commandes
  make up                - Démarrer
  make down              - Arrêter
  make logs              - Voir les logs
  make status            - État des services
